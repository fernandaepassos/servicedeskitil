        #region Retorna texto a partir de determinada posição
        /// <summary>
        /// Retorna texto a partir de determinada posição
        /// </summary>
        /// <param name="strTexto">Texto a ser pesquisado</param>
        /// <param name="intPosicao">Número inteiro com a posição do inicia da busca</param>
        /// <returns>Retorna o texto encontrado</returns>
        public string GetCaraterPorPosicao(string strTexto, int intPosicao)
        {
            strTexto = strTexto.Substring(strTexto.Length - (intPosicao-1), (intPosicao-1));
            return strTexto;
        }
        #endregion

O QUE: INCLUIR MÉTODO QUE RETORNA A POSIÇÃO DO CARACTER DENTRO DE UMA STRING

CONTEÚDO:

        #region Retorna posição do caracter
        /// <summary>
        /// Retorna posição do caracter
        /// </summary>
        /// <param name="strTexto">Texto onde será pesquisado</param>
        /// <param name="strCaracter">Caracter que será pesquisado a posição</param>
        /// <returns>Número inteiro indicando a posição</returns>
        public int GetPosicaoCaracter(string strTexto, string strCaracter)
        {
            int i = 1;
            int posic = 0;


            while (i < strTexto.Length)
            {
                if ((strTexto.Substring(strTexto.Length - i, 1)) == strCaracter)
                    return i;
                else
                    posic = 0;
                i++;
            }
            return posic;
        }
        #endregion
